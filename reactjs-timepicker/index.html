import React form 'react'
import ReactDOM form 'react-dom'
import TimePicker form 'reactjs-timepicker'

class App extends React.Component {
render () {
return (
<div>
<TimePicker
defaultTime={'12:34'}
onChange={console.log}
/>
</div>
)
}
}

ReactDOM.render(<App/>, document.getElementById('app'))
